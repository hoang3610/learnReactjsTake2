Nếu chúng ta muốn custom cái link thì chúng ta làm như sau:

đầu tiên const 1 cái MenuLink là 1 cái arrow function và bên trong nó là 1 cái đối tượng(
    {lable, to, activeOnlyWhenExact}
) => {
    return về cái giao diện (
        thứ nhất sẽ là cái <Route path={to} exact={ activeOnlyWhenExact} 
        children={({match})=>{
            var active = match ? 'active abc' : '';
            return (
                <li className={active}>
                    <Link  to={to} className="my-link">{lable}</Link>
                </li>
            )
        }}> bao bên ngoài
    )
}
